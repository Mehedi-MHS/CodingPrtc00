Use if() as like php's isset() option in js

   var x;//no value set
   if(x){
     alert("hello");
   } //returns false. so no alert is shown
   if(!x){
     alert("hello")
   }//returns true // hello
   
   var a="some content";//value set
   if(a){
     alert("hello");
   }//returns true as "a" have a value. Alert will be shown.
